## Example

```jsx preview
import {Provider} from '@clayui/core';
import TimePicker from '@clayui/time-picker';
import React from 'react';

export default function App() {
  const [state, setState] = React.useState({
    hours: '--',
    minutes: '--',
  });

  return (
    <Provider spritemap="/icons.svg">
      <div className="p-4">
        <TimePicker
          onChange={setState}
          timezone="GMT+01:00"
          use12Hours
          value={state}
        />
      </div>
    </Provider>
  );
}
```
